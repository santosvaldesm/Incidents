/*
	001. CONSULTA REGISTROS CUENTAS REASEGURO ECOSISTEMA
	se ejecuta en la instacia de PDN
*/

WITH REA_ENLACE_CUENTA 
AS (
  SELECT A.ROWID AS ROW_ID, A.*
  FROM TREA_ENLACE_CRW_MVTO A
  WHERE A.CDTIPO_MOVIMIENTO IN ('18','4')
  -- El siguiente filtro se usa tambien para saber si el movimiento se encuentra en la fecha para que se hace el cierre o para otra fecha
 AND A.FECIERRE BETWEEN TO_DATE('INITIAL_DATE' || '000000', 'YYYYMMDDHH24MISS') AND TO_DATE('END_DATE' || '235959', 'YYYYMMDDHH24MISS')    -- Estas son las fechas que hay que cambiar
 AND A.CDAPLICACION = '9012'
--AND A.CDTRANSACCION LIKE '648372-cc:5920442%'
-- AND NMSINIESTRO = '9180000124079' --AND A.CDTRANSACCION LIKE '637862%' 
--AND NMSINIESTRO = '9180000120582' AND A.CDTRANSACCION LIKE '657933-cc:5978784%'
--AND NMSINIESTRO = '9180000131122' AND A.CDTRANSACCION LIKE '658002-cc:5978852%'

)


SELECT A.ROW_ID 
 ,A.CDTRANSACCION
 ,A.CDAPLICACION
 ,A.FECIERRE
 ,A.CDRAMO
 ,A.CDPLAN
 ,A.CDMONEDA
 ,A.CDGARANTIA
 ,A.CDTIPO_POLIZA
 ,A.CDCONTRATO
 ,A.POCOMISION
 ,A.PODEPOSITO
 ,A.POINTERES
 ,A.POAJUSTE_COMISION
 ,A.POIMPUESTO
 ,A.PTGASTOS_VARIOS
 ,A.OPTIPO_VALOR
 ,A.CDTIPO_MOVIMIENTO
 ,CASE WHEN A.SNCORREDOR = 'N' THEN A.PTVALOR_MOVIMIENTO ELSE B.PTVALOR_MOVIMIENTO END PTVALOR_MOVIMIENTO
 ,CASE WHEN A.SNCORREDOR = 'N' THEN A.PTVALOR_MOV_CEDIDO ELSE B.PTVALOR_MOV_CEDIDO END PTVALOR_MOV_CEDIDO
 ,A.SNPROPORCIONAL
 ,A.SNFACULTATIVO
 ,A.CDCIA_SURA
 ,A.CDTIPO_CONTRATO
 ,A.OPTIPO_REASEGURO
 ,A.NMPOLIZA
 ,A.CDTOMADOR
 ,A.FEINI_VIGENCIA
 ,A.FEFIN_VIGENCIA
 ,CASE WHEN A.SNCORREDOR = 'N' THEN A.CDREASEGURADOR ELSE B.CDREASEGURADOR END CDREASEGURADOR
 ,A.SNCORREDOR
 ,B.CDCORREDOR
 ,CASE WHEN CASE WHEN A.SNCORREDOR = 'N' THEN A.CDREASEGURADOR ELSE B.CDREASEGURADOR END = '00000' THEN 'RETENIDO' ELSE 'CEDIDO' END TIPORA
 ,CASE WHEN A.CDTIPO_MOVIMIENTO = 18 THEN 'SINIESTRO LIQUIDADO' ELSE 'SALVAMENTOS' END TIPOMOVIMIENTO
 ,CASE WHEN A.CDMONEDA = 0 THEN 'COP' ELSE 'USD' END MONEDA
 ,A.NMSINIESTRO
 ,A.NMRECIBO
FROM REA_ENLACE_CUENTA A 
LEFT JOIN TREA_ENLACE_CORREDOR_MVTO B ON A.CDTRANSACCION = B.CDTRANSACCION
  AND A.FECIERRE = B.FECIERRE
  AND A.CDRAMO = B.CDRAMO
  AND A.CDGARANTIA = B.CDGARANTIA
  AND A.CDCONTRATO = B.CDCONTRATO
  AND A.CDTIPO_MOVIMIENTO = B.CDTIPO_MOVIMIENTO
  AND A.OPTIPO_REASEGURO = B.OPTIPO_REASEGURO
  AND A.NMPOLIZA = B.NMPOLIZA
  AND A.CDREASEGURADOR = B.CDCORREDOR
  AND A.SNCORREDOR = 'S'
ORDER BY CDTRANSACCION
  ,PTVALOR_MOVIMIENTO DESC
