	001. CONSULTA REGISTROS CUENTAS REASEGURO GW
	se ejecuta en la instacia de PDNGWCC

WITH 

TRANSACTIONS
AS (
   SELECT DECODE(PO.MASTERPOLICYNUMBER, NULL, PO.POLICYNUMBER, PO.MASTERPOLICYNUMBER) POLICYNUMBER 
    ,CL.CLAIMNUMBER
    ,CASE WHEN PO.POLICYTYPE = 10001 THEN '040' ELSE
      CASE WHEN TCM.EXTERNALCODE IS NULL THEN PO.LINE_EXT ELSE TCM.EXTERNALCODE END     
     END ACCOUNTING_BRANCH
    ,TR.PUBLICID
    ,TR.ID TRANSACTIONID
    ,TR.CREATETIME
    ,TRS.L_ES_CO TRANSACTIONSTATUS
    ,TR.STATUS 
    ,TRSB.L_EN_US AS TIPOTRANSACCION
    ,TR.RECOVERYCATEGORY
    ,TR.CHECKID
    ,IT.CLAIMAMOUNT
    ,TR.SUBTYPE
    ,TR.SAPDETAIL_EXT
    ,TR.ISDEVOLUTION_EXT
    ,0 ISREFLECTION
    ,CRR.TYPECODE MONEDA
  FROM ADM_GWCC.CC_TRANSACTION TR 
   INNER JOIN ADM_GWCC.CCTL_TRANSACTIONSTATUS TRS ON TR.STATUS = TRS.ID
   INNER JOIN ADM_GWCC.CC_EXPOSURE EXP ON EXP.ID = TR.EXPOSUREID
   INNER JOIN ADM_GWCC.CC_CLAIM CL ON CL.ID = EXP.CLAIMID
   INNER JOIN ADM_GWCC.CC_POLICY PO ON CL.POLICYID = PO.ID
   INNER JOIN ADM_GWCC.CC_TRANSACTIONLINEITEM IT ON IT.TRANSACTIONID = TR.ID
   INNER JOIN ADM_GWCC.CCTL_TRANSACTION TRSB ON TR.SUBTYPE = TRSB.ID 
   LEFT JOIN ADM_GWCC.CC_COVERAGE COV ON COV.ID = EXP.COVERAGEID
   LEFT JOIN ADM_GWCC.CCTL_COVERAGETYPE CTL ON CTL.ID = COV.TYPE
   LEFT JOIN ADM_GWCC.CCTL_OFFERINGTYPE_EXT OFT ON OFT.ID = COV.OFFERINGTYPE_EXT
   LEFT JOIN ADM_GWCC.CCX_TYPECODEMAPPER_EXT TCM ON
       TCM.LISTNAME = 'Line_Ext'
       AND TCM.NAMESPACE = 'AccountingLineFromCoverageOffering'
       AND TCM.INTERNALCODE = CTL.TYPECODE||'-'||OFT.TYPECODE
   LEFT JOIN ADM_GWCC.CCTL_CURRENCY CRR ON CRR.ID = TR.CURRENCY   
  WHERE TR.RETIRED = 0
    AND TR.PUBLICID NOT LIKE 'ctm%'
    AND CL.RETIRED = 0
    AND TR.SUBTYPE IN (1, 3)
    AND IT.CLAIMAMOUNT <> 0 
   
            
            --AND tr.publicid IN ('cc:4149996')    
),


PAYMENTS
AS (
   SELECT P.* FROM (
     SELECT DISTINCT TR.POLICYNUMBER
      ,TR.CLAIMNUMBER
      ,TR.ACCOUNTING_BRANCH
      ,CH.CHECKNUMBER AS CDTRANSACCION
      ,TR.PUBLICID
      ,TR.TRANSACTIONID
      ,'' REFLECTIONID
      ,TR.CREATETIME
      ,TRS.L_ES_CO AS REFERENCESTATUS
      ,CH.STATUS
      ,TR.TIPOTRANSACCION
      ,(SELECT COUNT(1) FROM ADM_GWCC.CCX_RETENTION_EXT RET WHERE RET.CHECKID = CH.ID) RETENTIONS
      , (CASE WHEN TR.CLAIMAMOUNT<0 THEN  SAPA.SAPACCOUNTINGDATE_EXT ELSE SAP.SAPACCOUNTINGDATE_EXT END) ACCOUNTINGDATE
      ,CASE WHEN NVL(CH.BULKINVOICEITEMINFOID, - 1) = - 1 THEN 'false' ELSE 'true' END BULKED
      ,TR.ISREFLECTION
      ,MONEDA
      ,0 AGREEMENTGROUPRI
      ,'' RF_STATUS
     FROM TRANSACTIONS TR 
      INNER JOIN ADM_GWCC.CC_CHECK CH ON CH.ID = TR.CHECKID
      INNER JOIN ADM_GWCC.CCTL_TRANSACTIONSTATUS TRS ON CH.STATUS = TRS.ID
      LEFT JOIN ADM_GWCC.CCX_SAPDETAIL_EXT SAP ON CH.SAPDETAIL_EXT = SAP.ID
      LEFT JOIN ADM_GWCC.CCX_SAPDETAIL_EXT SAPA ON CH.SAPDETAILNEGATIVE_EXT = SAPA.ID
     WHERE CH.RETIRED = 0
      AND TR.SUBTYPE = 1
      AND CH.STATUS NOT IN (5,9,12,15,18,10)
       AND (CASE WHEN TR.CLAIMAMOUNT<0 THEN  SAPA.SAPACCOUNTINGDATE_EXT ELSE SAP.SAPACCOUNTINGDATE_EXT END) 
            BETWEEN TO_DATE('INITIAL_DATE' || '000000', 'YYYYMMDDHH24MISS') AND TO_DATE('END_DATE' || '235959', 'YYYYMMDDHH24MISS') -- Estas son las fechas que hay que cambiar
            
   ) P
   WHERE (CASE WHEN P.STATUS = 7 AND P.BULKED = 'true' THEN 1 ELSE 0 END) = 0
),


RECOVERYS 
AS (
   SELECT DISTINCT TR.POLICYNUMBER
    ,TR.CLAIMNUMBER
    ,TR.ACCOUNTING_BRANCH
    ,TR.PUBLICID AS CDTRANSACCION
    ,TR.PUBLICID
    ,TR.TRANSACTIONID
    ,'' REFLECTIONID
    ,TR.CREATETIME
    ,CASE WHEN NVL(TR.ISDEVOLUTION_EXT, 0) = 1 AND TR.RECOVERYCATEGORY = 10001 THEN 'Devolucion' ELSE TR.TRANSACTIONSTATUS END AS REFERENCESTATUS
    ,TR.STATUS
    ,CASE WHEN TR.RECOVERYCATEGORY = 10001 THEN 'Salvage' ELSE TR.TIPOTRANSACCION END AS TIPOTRANSACCION
    ,999 RETENTIONS
    ,SAP.SAPACCOUNTINGDATE_EXT
    ,'false' BULKED   
    ,TR.ISREFLECTION
    ,MONEDA
    ,0 AGREEMENTGROUPRI
    ,'' RF_STATUS
   FROM TRANSACTIONS TR
         LEFT JOIN ADM_GWCC.CCX_SAPDETAIL_EXT SAP ON TR.SAPDETAIL_EXT = SAP.ID

   WHERE TR.SUBTYPE = 3
     AND TR.STATUS NOT IN (5,9,12,15,18,10)
      AND (CASE WHEN TR.SAPDETAIL_EXT IS NULL OR SAP.SAPACCOUNTINGDATE_EXT IS NULL THEN TR.CREATETIME ELSE SAP.SAPACCOUNTINGDATE_EXT END) 
            BETWEEN TO_DATE('INITIAL_DATE' || '000000', 'YYYYMMDDHH24MISS') AND TO_DATE('END_DATE' || '235959', 'YYYYMMDDHH24MISS') -- Estas son las fechas que hay que cambiar
),


TRANSACTIONS_ALL
AS (
   SELECT * FROM PAYMENTS
   UNION
   SELECT * FROM RECOVERYS
),

TR_REINSURANCE 
AS (
  SELECT AGRG.TRANSACTIONID 
   ,AGRI.AGREEMENTNUMBER
   ,AGRI.SUBTYPE
   ,RRI.PARTICIPATIONPERCENTAGE
   ,RRI.PARTICIPATIONAMOUNT
   ,NVL(RRI.DEPOSITRATE, 0) DEPOSITRATE
   ,NVL(RRI.INTERESTRATE, 0) INTERESTRATE
   ,NVL(RRI.COMMISSIONRATE, 0) COMMISSIONRATE
   ,NVL(RRI.COMMISSIONADJUSTMENTRATE, 0) COMMISSIONADJUSTMENTRATE
   ,CASE WHEN TR.SUBTYPE = 1 OR (TR.SUBTYPE = 3 AND TR.RECOVERYCATEGORY <> 10001) THEN 'CLAIM_AMOUNT' ELSE 
     CASE WHEN TR.SUBTYPE = 3 AND TR.RECOVERYCATEGORY = 10001 THEN 'SALVAGE_AMOUNT' ELSE '' END
    END TIPO
   ,NVL(CTC.SURACODE_EXT, '00000') REINSURERCODE 
   ,CTC.NAMEDENORM REINSURERNAME
   ,NVL(CTCB.SURACODE_EXT, '') BROKERCODE 
   ,NVL(CTCB.NAMEDENORM, '') BROKERNAME
  FROM ADM_GWCC.CCX_AGREEMENTGROUPRI_EXT AGRG
   INNER JOIN ADM_GWCC.CCX_AGREEMENTRI_EXT AGRI ON AGRG.ID = AGRI.AGREEMENTGROUPID
   INNER JOIN ADM_GWCC.CCX_REINSURERRI_EXT RRI ON RRI.AGREEMENTID = AGRI.ID
   INNER JOIN ADM_GWCC.CC_CLAIMCONTACT CCO ON RRI.REINSURERID = CCO.ID
   INNER JOIN ADM_GWCC.CC_CONTACT CTC ON CTC.ID = CCO.CONTACTID
   INNER JOIN ADM_GWCC.CC_TRANSACTION TR ON AGRG.TRANSACTIONID = TR.ID
   LEFT JOIN ADM_GWCC.CC_CLAIMCONTACT CCOB ON RRI.BROKERID = CCOB.ID
   LEFT JOIN ADM_GWCC.CC_CONTACT CTCB ON CTCB.ID = CCOB.CONTACTID
  WHERE AGRG.RETIRED = 0
    AND (
       AGRG.DISTRIBUTIONTYPE = 'PAYMENT' OR AGRG.DISTRIBUTIONTYPE = 'RECOVERY' OR AGRG.DISTRIBUTIONTYPE IS NULL
      )
    AND NVL(RRI.PARTICIPATIONAMOUNT, 0) <> 0  
    AND NVL(AGRG.REFLECTION, 0) = 0
    AND EXISTS (
        SELECT * FROM TRANSACTIONS_ALL T WHERE T.TRANSACTIONID = AGRG.TRANSACTIONID 
    )
),

TR_PURE_RETENTION
AS (
  SELECT AGRG.TRANSACTIONID
   ,'RP_' || '199RC040RE' AGREEMENTNUMBER
   ,'PureRetention' SUBTYPE
   ,100 AS PARTICIPATIONPERCENTAGE
   ,AGRG.RETENTIONAMOUNT AS PARTICIPATIONAMOUNT
   ,0 AS DEPOSITRATE
   ,0 AS INTERESTRATE
   ,0 AS COMMISSIONRATE
   ,0 AS COMMISSIONADJUSTMENTRATE
   ,'RETENTION_AMOUNT' AS TIPO
   ,'00000' REINSURERCODE 
   ,'seguros generales suramericana s.a.' AS REINSURERNAME
   ,'' BROKERCODE 
   ,'' BROKERNAME
  FROM ADM_GWCC.CCX_AGREEMENTGROUPRI_EXT AGRG
  WHERE AGRG.RETIRED = 0
  AND (
      AGRG.DISTRIBUTIONTYPE = 'PAYMENT'
      OR 
      AGRG.DISTRIBUTIONTYPE = 'RECOVERY'
      OR AGRG.DISTRIBUTIONTYPE IS NULL
      )
      AND NVL(AGRG.RETENTIONAMOUNT, 0) <> 0
      AND NVL(AGRG.REFLECTION, 0) = 0
      AND EXISTS (
        SELECT * FROM TRANSACTIONS_ALL T WHERE T.TRANSACTIONID = AGRG.TRANSACTIONID 
      )
  ),
  
TR_REINSURANCE_ALL 
AS (
  SELECT * FROM TR_REINSURANCE
  UNION
  SELECT * FROM TR_PURE_RETENTION
),


/* ********************************************************************************************************* */
/* *********************************************** REFLEJOS ************************************************ */
/* ********************************************************************************************************* */


REFLECTIONS 
AS (
   SELECT 
     DECODE(P.MASTERPOLICYNUMBER, NULL, P.POLICYNUMBER, P.MASTERPOLICYNUMBER) POLICYNUMBER
     ,C.CLAIMNUMBER 
     ,CASE WHEN P.POLICYTYPE = 10001 THEN '040' ELSE
       CASE WHEN TCM.EXTERNALCODE IS NULL THEN P.LINE_EXT ELSE TCM.EXTERNALCODE END     
      END ACCOUNTING_BRANCH
     ,CASE WHEN T.SUBTYPE = 1 THEN CH.CHECKNUMBER ELSE T.PUBLICID END CDTRANSACCION
     ,T.PUBLICID
     ,T.ID TRANSACTIONID
     ,R.PUBLICID REFLECTIONID
     ,CASE WHEN NVL(R.ID, 0) = 0 THEN AGRG.CREATETIME ELSE R.CREATETIME END CREATETIME
     ,CASE WHEN NVL(T.ISDEVOLUTION_EXT, 0) = 1 AND T.RECOVERYCATEGORY = 10001 THEN 'Devolucion' ELSE STTR.L_ES_CO END AS REFERENCESTATUS 
     ,NVL(CAST(R.SAPSTATUS AS NUMBER), -1) STATUS
     ,CASE WHEN T.RECOVERYCATEGORY = 10001 THEN 'Salvage' ELSE TRST.L_EN_US END AS TIPOTRANSACCION
     ,0 RETENTIONS
     ,CASE WHEN NVL(R.ID, 0) = 0 THEN AGRG.CREATETIME ELSE CASE WHEN R.SAPACCOUNTINGDATE IS NULL THEN R.CREATETIME ELSE R.SAPACCOUNTINGDATE END END AS ACCOUNTINGDATE
     ,'false' AS BULKED
     ,1 ISREFLECTION
     ,CRR.TYPECODE MONEDA
     ,AGRG.ID AGREEMENTGROUPRI
     ,CASE WHEN NVL(R.ID, 0) = 0 THEN 'SIN ENVIO A SAP' ELSE CASE WHEN R.SAPSTATUS = 0 THEN 'SIN RESPUESTA DE SAP' ELSE 'RECIBIDO SAP' END END RF_STATUS
     ,T.STATUS TRANSACTIONSTATUS
     ,T.SUBTYPE
     ,T.CHECKID
   FROM ADM_GWCC.CCX_AGREEMENTGROUPRI_EXT AGRG 
    LEFT JOIN ADM_GWCC.CCX_MESSAGINGREFLECTION_EXT R ON R.AGREEMENTGROUPRI = AGRG.ID
    INNER JOIN ADM_GWCC.CC_CLAIM C ON AGRG.CLAIMID = C.ID 
    INNER JOIN ADM_GWCC.CC_POLICY P ON C.POLICYID = P.ID
    INNER JOIN ADM_GWCC.CC_TRANSACTION T ON AGRG.TRANSACTIONID = T.ID
    INNER JOIN ADM_GWCC.CC_EXPOSURE EX ON T.EXPOSUREID = EX.ID
    INNER JOIN ADM_GWCC.CC_COVERAGE COV ON EX.COVERAGEID = COV.ID
    INNER JOIN ADM_GWCC.CCTL_COVERAGETYPE CTL ON COV.TYPE = CTL.ID
    LEFT JOIN ADM_GWCC.CCTL_OFFERINGTYPE_EXT OFT ON COV.OFFERINGTYPE_EXT = OFT.ID
    LEFT JOIN ADM_GWCC.CCX_TYPECODEMAPPER_EXT TCM ON TCM.LISTNAME = 'Line_Ext'
         AND TCM.NAMESPACE = 'AccountingLineFromCoverageOffering'
         AND TCM.INTERNALCODE = CTL.TYPECODE||'-'||OFT.TYPECODE
    INNER JOIN ADM_GWCC.CCTL_CURRENCY CRR ON T.CURRENCY = CRR.ID
    INNER JOIN ADM_GWCC.CCTL_COSTCATEGORY TLCOS ON T.COSTCATEGORY = TLCOS.ID 
    INNER JOIN ADM_GWCC.CCTL_TRANSACTION TRST ON T.SUBTYPE = TRST.ID
    INNER JOIN ADM_GWCC.CCTL_TRANSACTIONSTATUS STTR ON T.STATUS = STTR.ID 
    LEFT JOIN ADM_GWCC.CC_CHECK CH ON T.CHECKID = CH.ID
    LEFT JOIN ADM_GWCC.CCTL_CURRENCY CRR ON CRR.ID = T.CURRENCY
   WHERE 
    AGRG.RETIRED = 0 AND NVL(R.RETIRED, 0) = 0 -- reflejos y distribuciones no retiradas
    AND NVL(AGRG.REFLECTION, 0) = 1 -- distribuciones tipo reflejos
    AND T.SUBTYPE IN (1, 3) -- transacciones tipo pago y recupero
    AND AGRG.DISTRIBUTIONTYPE IN ('PAYMENT', 'RECOVERY') -- distribucion de reaseguro tipo pago y recupero
    AND (CASE WHEN P.POLICYTYPE = 10001 AND T.PUBLICID LIKE 'ctm%' THEN 0 ELSE 1 END) = 1 -- excluir transacciones migradas de Autos
    AND NVL(T.ISRSA_EXT, 0) = 0 -- excluir transacciones de RSA
    
    -- Filtro de fecha, si tiene tabla de reflejos por fecha de contabilizacion si la tiene, sino fecha de creacion, si no tiene tabla de reflejos fecha de creacion de la distribucion
    AND (CASE WHEN NVL(R.ID, 0) = 0 THEN AGRG.CREATETIME ELSE CASE WHEN R.SAPACCOUNTINGDATE IS NULL THEN R.CREATETIME ELSE R.SAPACCOUNTINGDATE END END)
      BETWEEN TO_DATE('INITIAL_DATE' || '000000', 'YYYYMMDDHH24MISS') AND TO_DATE('END_DATE' || '235959', 'YYYYMMDDHH24MISS')   -- Estas son las fechas que hay que cambiar
    
    --AND R.STATUSWORKQUEUE = 1
    --AND R.SAPSTATUS = 1
    --AND R.SAPREPLYDATE IS NOT NULL 
),

REFLECTIONS_FILT AS (
 SELECT DISTINCT R.*
 FROM REFLECTIONS R
  INNER JOIN ADM_GWCC.CC_CHECK CH ON R.CHECKID = CH.ID 
 WHERE R.SUBTYPE = 1
  AND CH.STATUS NOT IN (5,7) -- transacciones de cheques no denegados o detenidos que no se van a sap
  
 UNION ALL
 
 SELECT DISTINCT R.*
 FROM REFLECTIONS R
 WHERE R.SUBTYPE = 3
  AND R.TRANSACTIONSTATUS NOT IN (5,7) -- transacciones de recuperos no denegados o detenidos que no se van a sap
),

RF_REINSURANCE 
AS (
  SELECT AGRG.ID 
   ,AGRG.TRANSACTIONID 
   ,AGRI.AGREEMENTNUMBER
   ,AGRI.SUBTYPE
   ,RRI.PARTICIPATIONPERCENTAGE
   ,RRI.PARTICIPATIONAMOUNT
   ,NVL(RRI.DEPOSITRATE, 0) DEPOSITRATE
   ,NVL(RRI.INTERESTRATE, 0) INTERESTRATE
   ,NVL(RRI.COMMISSIONRATE, 0) COMMISSIONRATE
   ,NVL(RRI.COMMISSIONADJUSTMENTRATE, 0) COMMISSIONADJUSTMENTRATE
   ,CASE WHEN TR.SUBTYPE = 1 OR (TR.SUBTYPE = 3 AND TR.RECOVERYCATEGORY <> 10001) THEN 'CLAIM_AMOUNT' ELSE 
     CASE WHEN TR.SUBTYPE = 3 AND TR.RECOVERYCATEGORY = 10001 THEN 'SALVAGE_AMOUNT' ELSE '' END
    END TIPO
   ,NVL(CTC.SURACODE_EXT, '00000') REINSURERCODE 
   ,CTC.NAMEDENORM REINSURERNAME
   ,NVL(CTCB.SURACODE_EXT, '') BROKERCODE 
   ,NVL(CTCB.NAMEDENORM, '') BROKERNAME
  FROM ADM_GWCC.CCX_AGREEMENTGROUPRI_EXT AGRG
   INNER JOIN ADM_GWCC.CCX_AGREEMENTRI_EXT AGRI ON AGRG.ID = AGRI.AGREEMENTGROUPID
   INNER JOIN ADM_GWCC.CCX_REINSURERRI_EXT RRI ON RRI.AGREEMENTID = AGRI.ID
   INNER JOIN ADM_GWCC.CC_CLAIMCONTACT CCO ON RRI.REINSURERID = CCO.ID
   INNER JOIN ADM_GWCC.CC_CONTACT CTC ON CTC.ID = CCO.CONTACTID
   INNER JOIN ADM_GWCC.CC_TRANSACTION TR ON AGRG.TRANSACTIONID = TR.ID
   LEFT JOIN ADM_GWCC.CC_CLAIMCONTACT CCOB ON RRI.BROKERID = CCOB.ID
   LEFT JOIN ADM_GWCC.CC_CONTACT CTCB ON CTCB.ID = CCOB.CONTACTID
  WHERE AGRG.RETIRED = 0
    AND NVL(RRI.PARTICIPATIONAMOUNT, 0) <> 0  
    AND NVL(AGRG.REFLECTION, 0) = 1
    AND EXISTS (
        SELECT * FROM REFLECTIONS_FILT R WHERE R.AGREEMENTGROUPRI = AGRG.ID 
    )
),

RF_PURE_RETENTION
AS (
  SELECT AGRG.ID 
   ,AGRG.TRANSACTIONID
   ,'...RE' AGREEMENTNUMBER
   ,'PureRetention' SUBTYPE
   ,100 AS PARTICIPATIONPERCENTAGE
   ,AGRG.RETENTIONAMOUNT AS PARTICIPATIONAMOUNT
   ,0 AS DEPOSITRATE
   ,0 AS INTERESTRATE
   ,0 AS COMMISSIONRATE
   ,0 AS COMMISSIONADJUSTMENTRATE
   ,'RETENTION_AMOUNT' AS TIPO
   ,'00000' REINSURERCODE 
   ,'seguros generales suramericana s.a.' AS REINSURERNAME
   ,'' BROKERCODE 
   ,'' BROKERNAME
  FROM ADM_GWCC.CCX_AGREEMENTGROUPRI_EXT AGRG
  WHERE AGRG.RETIRED = 0
    AND NVL(AGRG.RETENTIONAMOUNT, 0) <> 0 
    AND NVL(AGRG.REFLECTION, 0) = 1
    AND EXISTS (
        SELECT * FROM REFLECTIONS_FILT R WHERE R.AGREEMENTGROUPRI = AGRG.ID 
    )
),

RF_REINSURANCE_ALL 
AS (
  SELECT * FROM RF_REINSURANCE
  UNION ALL
  SELECT * FROM RF_PURE_RETENTION
),

TRANSACTION_ALL_WITH_RF
AS (
   SELECT * FROM TRANSACTIONS_ALL
   
   UNION ALL
   
   SELECT POLICYNUMBER
    ,CLAIMNUMBER
    ,ACCOUNTING_BRANCH
    ,CDTRANSACCION
    ,PUBLICID
    ,TRANSACTIONID
    ,REFLECTIONID
    ,CREATETIME
    ,REFERENCESTATUS
    ,STATUS
    ,TIPOTRANSACCION
    ,RETENTIONS
    ,ACCOUNTINGDATE
    ,BULKED
    ,ISREFLECTION
    ,MONEDA
    ,AGREEMENTGROUPRI
    ,RF_STATUS 
   FROM REFLECTIONS
)
  
--, FINAL_ALL AS(

SELECT DISTINCT 
 T.POLICYNUMBER
 ,T.CLAIMNUMBER
 ,T.ACCOUNTING_BRANCH
 ,T.CDTRANSACCION
 ,T.PUBLICID
 ,T.REFLECTIONID
 ,T.CREATETIME
 ,CASE WHEN T.ISREFLECTION = 0 THEN RT.AGREEMENTNUMBER ELSE RF.AGREEMENTNUMBER END AGREEMENTNUMBER
 ,CASE WHEN T.ISREFLECTION = 0 THEN RT.SUBTYPE ELSE RF.SUBTYPE END SUBTYPE
 ,CASE WHEN T.ISREFLECTION = 0 THEN RT.REINSURERCODE || '-' || RT.REINSURERNAME ELSE RF.REINSURERCODE || '-' || RF.REINSURERNAME END REINSURER
 ,CASE WHEN T.ISREFLECTION = 0 THEN RT.PARTICIPATIONPERCENTAGE ELSE RF.PARTICIPATIONPERCENTAGE END PARTICIPATIONPERCENTAGE
 ,CASE WHEN T.ISREFLECTION = 0 THEN 
   CASE WHEN T.TIPOTRANSACCION = 'Salvage' THEN NVL(RT.PARTICIPATIONAMOUNT,0) * -1 ELSE NVL(RT.PARTICIPATIONAMOUNT,0) END
  ELSE  
   CASE WHEN T.TIPOTRANSACCION = 'Salvage' THEN NVL(RF.PARTICIPATIONAMOUNT,0) * -1 ELSE NVL(RF.PARTICIPATIONAMOUNT,0) END
  END PARTICIPATIONAMOUNT
 ,CASE WHEN T.ISREFLECTION = 0 THEN RT.DEPOSITRATE ELSE RF.DEPOSITRATE END DEPOSITRATE
 ,CASE WHEN T.ISREFLECTION = 0 THEN RT.INTERESTRATE ELSE RF.INTERESTRATE END INTERESTRATE
 ,CASE WHEN T.ISREFLECTION = 0 THEN RT.COMMISSIONRATE ELSE RF.COMMISSIONRATE END COMMISSIONRATE
 ,CASE WHEN T.ISREFLECTION = 0 THEN RT.COMMISSIONADJUSTMENTRATE ELSE RF.COMMISSIONADJUSTMENTRATE END COMMISSIONADJUSTMENTRATE
 ,CASE WHEN T.ISREFLECTION = 0 THEN RT.TIPO ELSE RF.TIPO END TIPO
 ,T.REFERENCESTATUS
 ,T.TIPOTRANSACCION
 ,T.ACCOUNTINGDATE
 ,T.BULKED
 ,CASE WHEN T.ISREFLECTION = 0 THEN RT.REINSURERCODE ELSE RF.REINSURERCODE END REINSURERCODE
 ,CASE WHEN T.ISREFLECTION = 0 THEN RT.BROKERNAME ELSE RF.BROKERNAME END BROKERNAME
 ,CASE WHEN T.ISREFLECTION = 0 THEN RT.BROKERCODE ELSE RF.BROKERCODE END BROKERCODE
 ,T.ISREFLECTION
 ,CASE WHEN CASE WHEN T.ISREFLECTION = 0 THEN RT.REINSURERCODE ELSE RF.REINSURERCODE END = '00000' THEN 'RETENIDO' ELSE 'CEDIDO' END TIPORA
 ,CASE WHEN T.TIPOTRANSACCION = 'Salvage' THEN 'SALVAMENTOS' ELSE 'SINIESTRO LIQUIDADO' END TIPOMOVIMIENTO
 ,MONEDA
 ,RF_STATUS
FROM TRANSACTION_ALL_WITH_RF T
 LEFT JOIN TR_REINSURANCE_ALL RT ON T.ISREFLECTION = 0 AND T.TRANSACTIONID = RT.TRANSACTIONID
 LEFT JOIN RF_REINSURANCE_ALL RF ON T.ISREFLECTION = 1 AND T.AGREEMENTGROUPRI = RF.ID
ORDER BY CLAIMNUMBER
  ,TIPO
  ,ACCOUNTINGDATE
  ,CDTRANSACCION
